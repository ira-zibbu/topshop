 //method to calculate discount, tax and final price
 public double calculateDiscount ( )
    {
        
        if (amount < 3000)
            discount = 0.0D;
        if (amount >= 3000 && amount < 4500)
            discount = amount*0.05;
        if (amount >= 4500 && amount < 6000)
            discount = amount*0.1;
        if (amount >= 6000)
            discount = amount*0.15;
        finalamount = amount - discount;
        return discount;
    }
